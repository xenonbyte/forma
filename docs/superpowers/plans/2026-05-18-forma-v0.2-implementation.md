```ts
  primary: "#3b82f6",
  background: "#FFFFFF",

export async function scanStyleDirectories(root: string): Promise<ScannedStyleDirectory[]> {
  const entries = await readdir(root, { withFileTypes: true });
  const styles: ScannedStyleDirectory[] = [];
  for (const entry of entries) {
    if (!entry.isDirectory() || entry.name.startsWith("_") || entry.name.startsWith(".")) {
      continue;
    }
    const designMdPath = join(root, entry.name, "DESIGN.md");
    if (await fileExists(designMdPath)) {
      styles.push({ name: entry.name, designMdPath });
    }
  }
  return styles.sort((left, right) => left.name.localeCompare(right.name));
}

export function extractVariablesFromDesignMd(source: string): StyleVariables {
  return styleVariablesSchema.parse({
    primary: matchColor(source, /primary[:\s]+(#[0-9a-fA-F]{6})/i) ?? defaultSyncVariables.primary,
    background: matchColor(source, /background[:\s]+(#[0-9a-fA-F]{6})/i) ?? defaultSyncVariables.background,
    "text-primary":
      matchColor(source, /text[- ]?primary[:\s]+(#[0-9a-fA-F]{6})/i) ??
      matchColor(source, /foreground[:\s]+(#[0-9a-fA-F]{6})/i) ??
      defaultSyncVariables["text-primary"],
    "font-heading": matchText(source, /heading[- ]?font[:\s]+["']?([^"'\n,]+)/i) ?? defaultSyncVariables["font-heading"],
    "font-body": matchText(source, /body[- ]?font[:\s]+["']?([^"'\n,]+)/i) ?? defaultSyncVariables["font-body"],
    "border-radius":
      matchNumber(source, /border[- ]?radius[:\s]+(\d+)/i) ??
      matchNumber(source, /corner[- ]?radius[:\s]+(\d+)/i) ??
      defaultSyncVariables["border-radius"],
    "spacing-unit":
      matchNumber(source, /spacing[- ]?unit[:\s]+(\d+)/i) ??
      matchNumber(source, /base[- ]?spacing[:\s]+(\d+)/i) ??
      defaultSyncVariables["spacing-unit"]
  });
}

export function classifyStyle(source: string): string {
  const text = source.toLowerCase();
  if (/\b(ai|llm|chat|assistant)\b/i.test(text)) return "AI 产品";
  if (/\b(tool|productivity|project|task)\b/i.test(text)) return "工具类";
  if (/\b(shop|commerce|retail|store)\b/i.test(text)) return "电商";
  if (/\b(finance|bank|payment|trading)\b/i.test(text)) return "金融";
  if (/\b(social|community|message)\b/i.test(text)) return "社交";
  if (/\b(health|medical|fitness)\b/i.test(text)) return "健康";
  return "其他";
}

export function describeStyle(source: string): string {
  return source
    .split(/\r?\n/)
    .map((line) => line.trim())
    .find((line) => line.length > 0 && !line.startsWith("#"))
    ?.slice(0, 50) ?? "Synced style";
}

export function sha256Hex(source: string | Buffer): string {
  return createHash("sha256").update(source).digest("hex");
}

function matchColor(source: string, pattern: RegExp): string | undefined {
  return source.match(pattern)?.[1];
}

function matchText(source: string, pattern: RegExp): string | undefined {
  return source.match(pattern)?.[1]?.trim();
}

function matchNumber(source: string, pattern: RegExp): string | undefined {
  return source.match(pattern)?.[1];
}

async function fileExists(file: string): Promise<boolean> {
  try {
    await access(file);
    return true;
  } catch (error) {
    if (error instanceof Error && "code" in error && error.code === "ENOENT") return false;
    throw error;
  }
}
```

Add this class shape to `packages/core/src/sync.ts` below helper functions:

```ts
export interface SyncServiceOptions {
  home: string;
  pencilService: PencilService;
  runner: CommandRunner;
  autoRun?: boolean;
  now?: () => number;
}

const staleRunningMs = 10 * 60 * 1000;

export class SyncService {
  private readonly home: string;
  private readonly pencilService: PencilService;
  private readonly runner: CommandRunner;
  private readonly autoRun: boolean;
  private readonly now: () => number;
  private readonly stateFile: string;

  constructor(options: SyncServiceOptions) {
    this.home = options.home;
    this.pencilService = options.pencilService;
    this.runner = options.runner;
    this.autoRun = options.autoRun ?? true;
    this.now = options.now ?? Date.now;
    this.stateFile = join(options.home, "sync-state.yaml");
  }

  async startSync(): Promise<{ task_id: string }> {
    await this.recoverFromCrash();
    await this.pencilService.checkAvailability();
    await this.checkGit();
    const current = await this.readStatus();
    if (current.status === "running") {
      throw new FormaError("SYNC_ALREADY_RUNNING", "A sync task is already in progress", { task_id: current.task_id });
    }

    const taskId = `sync-${randomBytes(4).toString("hex")}`;
    await this.writeStatus({
      status: "running",
      task_id: taskId,
      started_at: new Date(this.now()).toISOString(),
      progress: { phase: "git_clone", current: 0, total: 0 }
    });

    if (this.autoRun) {
      void this.runTask(taskId).catch(() => undefined);
    }

    return { task_id: taskId };
  }

  async getStatus(): Promise<SyncStatus> {
    await this.recoverFromCrash();
    return this.readStatus();
  }

  async recoverFromCrash(): Promise<void> {
    const status = await this.readStatus();
    if (status.status !== "running") return;
    if (this.now() - Date.parse(status.started_at) <= staleRunningMs) return;
    await this.writeStatus({
      status: "failed",
      task_id: status.task_id,
      error: { phase: "cleanup", message: "Sync task did not finish before crash recovery timeout" }
    });
  }

  private async checkGit(): Promise<void> {
    try {
      await this.runner.run("git", ["--version"], { timeoutMs: 5_000 });
    } catch (error) {
      throw new FormaError("SYNC_GIT_NOT_FOUND", "Git is not installed", { cause: errorMessage(error) });
    }
  }

  private async readStatus(): Promise<SyncStatus> {
    if (!(await fileExists(this.stateFile))) return { status: "idle" };
    return readYamlAs(this.stateFile, syncStatusSchema);
  }

  private async writeStatus(status: SyncStatus): Promise<void> {
    await writeYamlAtomic(this.stateFile, status);
  }

  private async runTask(taskId: string): Promise<void> {
    await this.writeStatus({
      status: "failed",
      task_id: taskId,
      error: { phase: "git_clone", message: "Sync task execution stopped before clone" }
    });
  }
}

function errorMessage(error: unknown): string {
  return error instanceof Error ? error.message : String(error);
}
```

```ts
    const existing = await this.readExistingStyles();
    const prepared: PreparedStyle[] = [];
    for (const [index, item] of scanned.entries()) {
      await this.updateProgress(taskId, "extracting_variables", index + 1, scanned.length, item.name);
      const designMd = await readFile(item.designMdPath, "utf8");
      const hash = sha256Hex(designMd);
      prepared.push({

        designMd,
        hash,

          name: item.name,
          description: describeStyle(designMd),
          design_md_path: `styles/${item.name}/DESIGN.md`,
          variables: extractVariablesFromDesignMd(designMd)
        },
        added: !existing.has(item.name),
        updated: existing.get(item.name) !== undefined && existing.get(item.name) !== hash
      });
    }

    let stylesFailed = 0;
    for (let index = 0; index < prepared.length; index += 5) {
      const batch = prepared.slice(index, index + 5);
      const rendered = await this.renderPreviewBatch(taskId, taskDir, batch, index, prepared.length);
      stylesFailed += batch.length - rendered;

    await this.updateProgress(taskId, "updating_index", prepared.length, prepared.length);
    await this.persistStyles(prepared);
    await this.writeStatus({
      status: "idle",
      last_sync: {
        completed_at: new Date(this.now()).toISOString(),
        styles_total: prepared.length,
        styles_added: prepared.filter((style) => style.added).length,
        styles_updated: prepared.filter((style) => style.updated).length,
```

```ts
  edges: GraphEdgeLayout[];
}

const nodeWidth = 120;
const nodeHeight = 60;

export function createNavigationLayout(nodes: GraphNodeInput[], edges: GraphEdgeInput[]): NavigationLayout {
  const size = canvasSize(nodes.length);
  if (nodes.length === 0) return { ...size, nodes: [], edges: [] };
  if (nodes.length === 1) {
    const [node] = nodes;
    return {
      ...size,
      nodes: [{ id: node.id, name: node.name, features: countFeatures(node.features), rawFeatures: node.features, x: size.width / 2, y: size.height / 2 }],
      edges: []
    };
  }

  const ids = new Set(nodes.map((node) => node.id));
  const layoutEdges = edges
    .filter((edge) => ids.has(edge.from) && ids.has(edge.to))
    .map((edge) => ({ from: edge.from, to: edge.to, trigger: edge.trigger ?? "No label" }));
  const positioned = initialCircle(nodes, size.width, size.height);
  const velocities = new Map(positioned.map((node) => [node.id, { vx: 0, vy: 0 }]));

  for (let iteration = 0; iteration < 100; iteration += 1) {
    applyRepulsion(positioned, velocities);
    applyAttraction(positioned, layoutEdges, velocities);
    for (const node of positioned) {
      const velocity = velocities.get(node.id)!;
      velocity.vx *= 0.9;
      velocity.vy *= 0.9;
      node.x = clamp(node.x + velocity.vx, nodeWidth / 2, size.width - nodeWidth / 2);
      node.y = clamp(node.y + velocity.vy, nodeHeight / 2, size.height - nodeHeight / 2);
    }
  }

  return { ...size, nodes: positioned, edges: layoutEdges };
}

export function countFeatures(features: string): number {
  return features.split("+").map((part) => part.trim()).filter(Boolean).length;
}
```

Add helper functions `canvasSize`, `initialCircle`, `applyRepulsion`, `applyAttraction`, `findNode`, and `clamp` in the same file. Use constants from the spec: repulsion `500`, attraction `0.01`, damping `0.9`, 100 iterations, min canvas `600x400`, add `200x200` per 5 nodes, max `1600x1200`.

```ts
import { mkdtemp, readFile } from "node:fs/promises";
import { tmpdir } from "node:os";
import { join, resolve } from "node:path";
import { createFormaStore } from "@xenonbyte/forma-core";

async function main() {
  const home = await mkdtemp(join(tmpdir(), "forma-live-sync-"));
  const store = createFormaStore({ home, bundledStylesDir: resolve("styles") });
  await store.styles.installBuiltInStyles();
  await store.sync.startSync();

  let status = await store.sync.getStatus();
  const deadline = Date.now() + 5 * 60 * 1000;
  while (status.status === "running" && Date.now() < deadline) {
    await new Promise((resolveTick) => setTimeout(resolveTick, 2000));
    status = await store.sync.getStatus();
  }

  if (status.status !== "idle") {
    throw new Error(`Live style sync failed: ${JSON.stringify(status)}`);
  }

  const styles = await store.styles.listStyles();
  if (styles.length === 0) {
    throw new Error("Live style sync produced no styles");
  }
  const preview = await readFile(join(home, styles[0].design_md_path.replace(/DESIGN\.md$/, "preview@2x.png")));
  if (preview.length === 0) {
    throw new Error(`Live style sync preview is empty for ${styles[0].name}`);
  }
  console.log(`Live style sync passed with ${styles.length} styles in ${home}`);
}

main().catch((error: unknown) => {
  const message = error instanceof Error ? error.stack ?? error.message : String(error);
  console.error(message);
  process.exitCode = 1;
});
```
